create Table Buch(
	name varchar(255) PRIMARY KEY,
    	genre varchar(20),
	ausgewählt bool,
	path varchar(255)
);

create Table Termine(
	name varchar(50) PRIMARY KEY,
    	datum date,
	uhrzeit time,
	beschreibung varchar(100),
	ort varchar(50),
	hinweis varchar(100)
);

create Table Pillen(
	name varchar(50) PRIMARY KEY,
	montag bool,
	dienstag bool,
	mittwoch bool,
	donnerstag bool,
	freitag bool,
	samstag bool,
	sonntag bool,
	zeit time
);
